\begin{align}
    y & = ax + b,    \\
    z & = x^2 + y^2.
\end{align}

また，スニペットを用意しているので，下記のように入力後タブキーを押すことでよく使う定型コードを呼び出せます．
\begin{itemize}
    \item \texttt{sec} や \texttt{ssec} で見出しを追加できる
    \item \texttt{eq} や \texttt{ali} で数式環境を展開できる
    \item \texttt{fig} や \texttt{btab} で図表を挿入できる
    \item \texttt{cite} や \texttt{eqr} で参照を追加できる
\end{itemize}

\section{定理環境の例}
\begin{definition}
    区間 \(I\) 上の関数 \(f\) が，任意の \(x,y\in I\) と \(0 \le \lambda \le 1\) に対して次を満たすとします．
    \[
        f(\lambda x + (1-\lambda)y) \le \lambda f(x) + (1-\lambda)f(y)
    \]
    このとき，\(f\) は凸であるといいます．
\end{definition}

\begin{theorem}\label{thm:supporting-line}
    微分可能な凸関数 \(f\) に対して，任意の \(x,y\in I\) で次が成り立ちます．
    \[
        f(y) \ge f(x) + f'(x)(y-x)
    \]
\end{theorem}

\begin{proof}
    凸性の定義を \(y=x+h\) に適用し，\(h \to 0\) の極限を考えると従います．
\end{proof}
